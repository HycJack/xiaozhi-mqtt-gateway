```javascript
handleConnect(connectData) {
    this.clientId = connectData.clientId;
    this.username = connectData.username;
    this.password = connectData.password;
    
    const parts = this.clientId.split('@@@');
    if (parts.length === 3) { // GID_test@@@mac_address@@@uuid
        try {
            const validated = validateMqttCredentials(this.clientId, this.username, this.password);
            this.groupId = validated.groupId;
            this.macAddress = validated.macAddress;
            this.uuid = validated.uuid;
            this.userData = validated.userData;
        } catch (error) {
            debug('认证失败:', error.message);
            this.close();
            return;
        }
    } else if (parts.length === 2) { // GID_test@@@mac_address
        this.groupId = parts[0];
        this.macAddress = parts[1].replace(/_/g, ':');
        if (!MacAddressRegex.test(this.macAddress)) {
            debug('无效的 macAddress:', this.macAddress);
            this.close();
            return;
        }
    } else {
        debug('无效的 clientId:', this.clientId);
        this.close();
        return;
    }
    this.replyTo = `devices/p2p/${parts[1]}`;
    
    this.server.addConnection(this);
    this.initializeDeviceTools();
}
```

## 四、示例：扩展 WebRTC 作为新传输协议

### 1. 实现 WebRTC 连接管理

```javascript
class WebRTCHandler {
    constructor(connection, options) {
        this.connection = connection;
        this.options = options;
        this.peerConnection = null;
        this.dataChannel = null;
        this.setupConnection();
    }

    setupConnection() {
        // 创建 PeerConnection
        this.peerConnection = new RTCPeerConnection({
            iceServers: [
                { urls: 'stun:stun.l.google.com:19302' }
            ]
        });

        // 创建数据通道
        this.dataChannel = this.peerConnection.createDataChannel('audio');
        this.dataChannel.onopen = () => {
            this.isConnected = true;
        };
        this.dataChannel.onmessage = (event) => {
            this.onData(event.data);
        };

        // 生成 offer
        this.peerConnection.createOffer()
            .then(offer => this.peerConnection.setLocalDescription(offer))
            .then(() => {
                // 发送 offer 到设备
                this.connection.sendMqttMessage(JSON.stringify({
                    type: 'webrtc_offer',
                    sdp: this.peerConnection.localDescription.sdp
                }));
            });
    }

    handleAnswer(answer) {
        // 处理设备的 answer
        this.peerConnection.setRemoteDescription(new RTCSessionDescription({
            type: 'answer',
            sdp: answer.sdp
        }));
    }

    sendData(data, timestamp) {
        if (this.dataChannel && this.dataChannel.readyState === 'open') {
            // 发送音频数据
            this.dataChannel.send(data);
        }
    }

    onData(data) {
        // 处理从设备接收的数据
        this.connection.bridge.sendAudio(data, Date.now());
    }

    close() {
        if (this.peerConnection) {
            this.peerConnection.close();
            this.peerConnection = null;
        }
    }
}
```

## 五、扩展 OAuth 2.0 认证协议

### 1. 实现 OAuth 2.0 认证

```javascript
function validateOAuth2Credentials(username, password, macAddress) {
    // 解析 OAuth 2.0 令牌
    const token = username.replace('oauth2:', '');
    
    // 验证令牌
    try {
        const decoded = jwt.verify(token, process.env.JWT_SECRET);
        
        // 检查令牌是否过期
        if (decoded.exp < Date.now() / 1000) {
            throw new Error('令牌已过期');
        }
        
        // 检查设备是否匹配
        if (decoded.mac_address !== macAddress) {
            throw new Error('设备不匹配');
        }
        
        return {
            groupId: decoded.group_id || 'GID_test',
            macAddress: macAddress,
            uuid: decoded.uuid || null,
            userData: {
                auth_type: 'oauth2',
                access_token: token,
                expires_in: decoded.exp - Date.now() / 1000
            }
        };
    } catch (error) {
        throw new Error('OAuth 2.0 认证失败: ' + error.message);
    }
}
```

### 2. 令牌刷新机制

```javascript
class OAuth2TokenManager {
    constructor(config) {
        this.clientId = config.client_id;
        this.clientSecret = config.client_secret;
        this.tokenEndpoint = config.token_endpoint;
        this.accessToken = null;
        this.expiresAt = null;
    }

    async refreshToken() {
        const response = await fetch(this.tokenEndpoint, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/x-www-form-urlencoded'
            },
            body: new URLSearchParams({
                grant_type: 'client_credentials',
                client_id: this.clientId,
                client_secret: this.clientSecret
            })
        });

        const data = await response.json();
        this.accessToken = data.access_token;
        this.expiresAt = Date.now() + data.expires_in * 1000;
        
        return this.accessToken;
    }

    async getAccessToken() {
        if (!this.accessToken || this.expiresAt <= Date.now()) {
            return await this.refreshToken();
        }
        return this.accessToken;
    }
}
```

### 1. 实现 JWT 认证

```javascript
function validateJWTCredentials(username, password, macAddress) {
    // 解析 JWT 令牌
    const token = username.replace('jwt:', '');
    
    try {
        const decoded = jwt.verify(token, process.env.JWT_SECRET, {
            algorithms: ['HS256']
        });
        
        // 验证令牌内容
        if (!decoded.mac_address || decoded.mac_address !== macAddress) {
            throw new Error('设备不匹配');
        }
        
        if (!decoded.iss || decoded.iss !== 'xiaozhi-mqtt') {
            throw new Error('无效的签发者');
        }
        
        if (!decoded.aud || decoded.aud !== 'mqtt-gateway') {
            throw new Error('无效的受众');
        }
        
        return {
            groupId: decoded.group_id || 'GID_test',
            macAddress: macAddress,
            uuid: decoded.uuid || null,
            userData: {
                auth_type: 'jwt',
                access_token: token,
                expires_in: decoded.exp - Date.now() / 1000,
                permissions: decoded.permissions || []
            }
        };
    } catch (error) {
        throw new Error('JWT 认证失败: ' + error.message);
    }
}
```

### 2. JWT 令牌生成

```javascript
function generateJWTToken(macAddress, options = {}) {
    const payload = {
        mac_address: macAddress,
        group_id: options.groupId || 'GID_test',
        uuid: options.uuid || null,
        permissions: options.permissions || ['voice', 'text'],
        iss: 'xiaozhi-mqtt',
        aud: 'mqtt-gateway',
        iat: Math.floor(Date.now() / 1000),
        exp: Math.floor(Date.now() / 1000) + (options.expiresIn || 3600)
    };
    
    return jwt.sign(payload, process.env.JWT_SECRET, {
        algorithm: 'HS256'
    });
}
```
